① 스프레드시트에 시트 추가

기존 학생명단 스프레드시트에 아래 이름의 시트를 추가하세요:
출결기록 청소배정 청소인증 상담슬롯 상담기록 익명제보 일기 누가기록 학생프로필 공지사항
→ 시트를 직접 만들 필요 없이 앱에서 데이터를 입력하면 자동 생성됩니다.

② Apps Script 코드 붙여넣기

스프레드시트 메뉴 → 확장 프로그램 → Apps Script 열기 후 아래 코드 전체를 붙여넣으세요.

// ★ Apps Script에 붙여넣고 저장 후 [배포 → 새 배포 → 웹앱] 으로 배포하세요.
// 실행 계정: 나 / 액세스: 모든 사용자

// ──────────────────────────────────────────
// CORS 허용 헤더 (모든 응답에 공통 적용)
// ──────────────────────────────────────────
function respond(obj) {
  return ContentService
    .createTextOutput(JSON.stringify(obj))
    .setMimeType(ContentService.MimeType.JSON);
}

// ──────────────────────────────────────────
// doGet: 읽기(action=sheet) + 쓰기 폴백(action=write)
// ──────────────────────────────────────────
function doGet(e) {
  const action = (e.parameter.action || '').trim();

  // 0) ping 테스트
  if (action === 'ping') {
    return respond({ ok: true, msg: 'Smart Class Manager GAS v2' });
  }

  // 1) 시트 읽기
  if (action === 'sheet') {
    const name = e.parameter.name || '';
    const ss = SpreadsheetApp.getActiveSpreadsheet();
    const sh = ss.getSheetByName(name);
    if (!sh) return respond({ error: '시트 없음: ' + name });
    return respond(sh.getDataRange().getValues());
  }

  // 2) 쓰기 (GET 방식 폴백 — 브라우저에서 POST 불가 시)
  if (action === 'write') {
    try {
      const data = JSON.parse(decodeURIComponent(e.parameter.data || '{}'));
      return writeData(data);
    } catch(err) {
      return respond({ ok: false, error: err.message });
    }
  }

  // 3) 학생명단 읽기 (하위 호환)
  if (action === 'read') {
    const ss = SpreadsheetApp.getActiveSpreadsheet();
    const sh = ss.getSheetByName('학생명단');
    if (!sh) return respond({ error: '학생명단 시트 없음' });
    return respond(sh.getDataRange().getValues());
  }

  return respond({ ok: true, msg: 'Smart Class Manager GAS v2' });
}

// ──────────────────────────────────────────
// doPost: 쓰기 (POST 방식)
// ──────────────────────────────────────────
function doPost(e) {
  try {
    const raw = e.postData ? e.postData.contents : '{}';
    const data = JSON.parse(raw);
    return writeData(data);
  } catch(err) {
    return respond({ ok: false, error: err.message });
  }
}

// ──────────────────────────────────────────
// writeData: 타입별 시트에 기록
// ──────────────────────────────────────────
function writeData(data) {
  const ss  = SpreadsheetApp.getActiveSpreadsheet();
  const now = new Date().toLocaleString('ko-KR');
  const type = data.type || '';

  // ── 출결 ──
  if (type === '출결') {
    const sh = getOrCreate(ss, '출결기록',
      ['날짜','학번','이름','상태','시간','사유','기록일시']);
    // 같은 날짜+학번 행이 있으면 업데이트, 없으면 추가 (중복 방지)
    const vals = sh.getDataRange().getValues();
    const dateStr = String(data.date || '');
    const numStr  = String(data.num  || '');
    const normDate = dateStr.slice(0,10).replace(/\./g,'-');
    // 저장할 날짜는 항상 YYYY-MM-DD 문자열로 통일
    const saveDate = normDate;
    for (let i = 1; i < vals.length; i++) {
      // Google Sheets에서 날짜 셀은 Date 객체로 반환됨 → Utilities.formatDate로 변환
      let rowDateRaw = vals[i][0];
      let rowDate;
      if (rowDateRaw instanceof Date) {
        rowDate = Utilities.formatDate(rowDateRaw, Session.getScriptTimeZone(), 'yyyy-MM-dd');
      } else {
        rowDate = String(rowDateRaw || '').slice(0,10).replace(/\./g,'-');
      }
      const rowNum = String(vals[i][1] || '');
      if (rowDate === normDate && rowNum === numStr) {
        sh.getRange(i+1, 1, 1, 7).setValues([[
          saveDate, data.num, data.name,
          data.status, data.time, data.reason||'', now
        ]]);
        return respond({ ok: true });
      }
    }
    sh.appendRow([saveDate, data.num, data.name,
      data.status, data.time, data.reason||'', now]);
    return respond({ ok: true });
  }

  // ── 청소 인증 ──
  if (type === '청소인증') {
    const sh = getOrCreate(ss, '청소인증',
      ['날짜','학번','이름','청소내용','기록일시']);
    sh.appendRow([data.date, data.num, data.name, data.content||'', now]);
    return respond({ ok: true });
  }

  // ── 청소 배정 (선생님 → 학생들이 읽음) ──
  if (type === '청소배정') {
    const sh = getOrCreate(ss, '청소배정', ['날짜','studentIds']);
    // 같은 날짜 행 있으면 업데이트, 없으면 추가
    // ★ 시트 날짜 셀은 Date 객체이므로 문자열로 변환 후 비교
    const vals = sh.getDataRange().getValues();
    const tz = Session.getScriptTimeZone();
    for (let i = 1; i < vals.length; i++) {
      let rowDate = vals[i][0];
      if (rowDate instanceof Date) {
        rowDate = Utilities.formatDate(rowDate, tz, 'yyyy-MM-dd');
      } else {
        rowDate = String(rowDate || '').slice(0,10).replace(/\./g,'-');
      }
      if (rowDate === String(data.date).slice(0,10)) {
        sh.getRange(i+1, 2).setValue(data.studentIds||'[]');
        return respond({ ok: true });
      }
    }
    sh.appendRow([data.date, data.studentIds||'[]']);
    return respond({ ok: true });
  }

  // ── 상담 슬롯 등록 (선생님 → 학생들이 읽음) ──
  if (type === '상담슬롯등록') {
    const sh = getOrCreate(ss, '상담슬롯',
      ['id','time','bookedBy','bookedNum','상태','기록일시']);
    sh.appendRow([data.slotId, data.time, '', '', '대기', now]);
    return respond({ ok: true });
  }

  // ── 상담 예약/취소 (학생 → 선생님이 읽음) ──
  if (type === '상담예약') {
    const sh = getOrCreate(ss, '상담슬롯',
      ['id','time','bookedBy','bookedNum','상태','기록일시']);
    const vals = sh.getDataRange().getValues();
    for (let i = 1; i < vals.length; i++) {
      if (String(vals[i][0]) === String(data.slotId)) {
        if (data.action === 'book') {
          sh.getRange(i+1, 3).setValue(data.name||'');
          sh.getRange(i+1, 4).setValue(data.num||'');
          sh.getRange(i+1, 5).setValue('예약완료');
        } else {
          sh.getRange(i+1, 3).setValue('');
          sh.getRange(i+1, 4).setValue('');
          sh.getRange(i+1, 5).setValue('대기');
        }
        return respond({ ok: true });
      }
    }
    // 슬롯 행이 없으면 새로 추가
    sh.appendRow([data.slotId, data.time||'',
      data.action==='book'?(data.name||''):'',
      data.action==='book'?(data.num||''):'',
      data.action==='book'?'예약완료':'대기', now]);
    return respond({ ok: true });
  }

  // ── 상담 슬롯 삭제 (미예약 슬롯 교사가 삭제) ──
  if (type === '상담슬롯삭제') {
    const sh = getOrCreate(ss, '상담슬롯',
      ['id','time','bookedBy','bookedNum','상태','기록일시']);
    const vals = sh.getDataRange().getValues();
    for (let i = 1; i < vals.length; i++) {
      if (String(vals[i][0]) === String(data.slotId)) {
        sh.deleteRow(i+1);
        return respond({ ok: true });
      }
    }
    return respond({ ok: true });
  }

  // ── 상담 완료 처리 (앱에서 삭제, 시트에는 완료 상태로 유지) ──
  if (type === '상담슬롯완료') {
    const sh = getOrCreate(ss, '상담슬롯',
      ['id','time','bookedBy','bookedNum','상태','기록일시']);
    const vals = sh.getDataRange().getValues();
    for (let i = 1; i < vals.length; i++) {
      if (String(vals[i][0]) === String(data.slotId)) {
        sh.getRange(i+1, 5).setValue('상담완료');
        sh.getRange(i+1, 6).setValue(now);
        return respond({ ok: true });
      }
    }
    return respond({ ok: true });
  }

  // ── 상담 기록 (상담 완료 후 내용 기재 → 상담기록 시트에 영구 보존) ──
  if (type === '상담기록') {
    const sh = getOrCreate(ss, '상담기록',
      ['상담날짜','상담시간','학번','학생이름','상담내용','기록일시']);
    sh.appendRow([data.date, data.slotTime||'',
      data.studentNum||'', data.studentName||'', data.content||'', now]);
    return respond({ ok: true });
  }

  // ── 익명 제보 ──
  if (type === '익명제보') {
    const sh = getOrCreate(ss, '익명제보',
      ['날짜','시간','내용','기록일시']);
    sh.appendRow([data.date, data.time, data.content||'', now]);
    return respond({ ok: true });
  }

  // ── 일기 ──
  if (type === '일기') {
    const sh = getOrCreate(ss, '일기',
      ['날짜','학번','이름','제목','내용','기록일시']);
    sh.appendRow([data.date, data.num, data.name,
      data.title||'', data.content||'', now]);
    return respond({ ok: true });
  }

  // ── 누가기록 ──
  if (type === '누가기록') {
    const sh = getOrCreate(ss, '누가기록',
      ['날짜','학번','이름','영역','관찰기록','AI초안','기록일시']);
    sh.appendRow([data.date, data.num, data.name,
      data.cat||'', data.content||'', data.draft||'', now]);
    return respond({ ok: true });
  }

  // ── 학생 추가 프로필 ──
  if (type === '학생프로필') {
    const sh = getOrCreate(ss, '학생프로필',
      ['studentId','학번','이름','진학예정','희망진로','MBTI','역할','선생님께','이모지']);
    // 같은 studentId 있으면 업데이트
    const vals = sh.getDataRange().getValues();
    for (let i = 1; i < vals.length; i++) {
      if (String(vals[i][0]) === String(data.studentId)) {
        sh.getRange(i+1, 1, 1, 9).setValues([[
          data.studentId, data.num||'', data.name||'',
          data.school||'', data.career||'', data.mbti||'',
          data.role||'', data.msg||'', data.emoji||''
        ]]);
        return respond({ ok: true });
      }
    }
    sh.appendRow([data.studentId, data.num||'', data.name||'',
      data.school||'', data.career||'', data.mbti||'',
      data.role||'', data.msg||'', data.emoji||'']);
    return respond({ ok: true });
  }

  // ── 공지사항 (추가/수정/삭제) ──
  if (type === '공지') {
    const sh = getOrCreate(ss, '공지사항',
      ['noticeId','date','title','content','action','기록일시']);
    const action = data.action || 'add';
    const vals = sh.getDataRange().getValues();
    // 같은 noticeId 있으면 업데이트
    for (let i = 1; i < vals.length; i++) {
      if (String(vals[i][0]) === String(data.noticeId)) {
        sh.getRange(i+1, 1, 1, 6).setValues([[
          data.noticeId, data.date||'', data.title||'',
          data.content||'', action, now
        ]]);
        return respond({ ok: true });
      }
    }
    sh.appendRow([data.noticeId, data.date||'', data.title||'',
      data.content||'', action, now]);
    return respond({ ok: true });
  }

  return respond({ ok: false, error: '알 수 없는 type: ' + type });
}

// ──────────────────────────────────────────
// 헬퍼: 시트 없으면 생성 + 헤더 추가
// ──────────────────────────────────────────
function getOrCreate(ss, name, headers) {
  let sh = ss.getSheetByName(name);
  if (!sh) {
    sh = ss.insertSheet(name);
    sh.appendRow(headers);
  } else if (sh.getLastRow() === 0) {
    sh.appendRow(headers);
  }
  return sh;
}
